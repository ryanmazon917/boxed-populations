World Population Split Into 100 Boxes¶

Note: The plot created by this code is not entirely accurate due to the data being very American/Euro-centric. Accurate gridded world population data to the degree needed for accuracy in this case is very large and expensive. I used the data I could find, so this is simply an educational exercise.¶

*For reference, this is a visualization of the city data that I used, from simplemaps.com. Clearly, this data does not capture the full picture of world population.

#loops through latitude and finds each value
#that fills the aggregate world population
#to 10%.
world_pop = filtered_df['population'].sum()
block_size = world_pop/10
currentsum = 0
index = 0
lat_vals = []
indvals = []

for x in range(9):
    currentsum = 0
    while currentsum < block_size:
        if currentsum >= 0:
            currentsum = currentsum + filtered_df['population'].iloc[index]
            index = index + 1
        else:
            currentsum['population'].add(filtered_df['population'].iloc[index])
            index = index + 1
    lat_vals.append(filtered_df['lat'].iloc[index])

#creating 10 quantiles of longitude values
#for each quantile of latitude values
lng_bins = []
lng_bins_edges = []

for i in range(10):
    bins, edges = pd.qcut(sorteddf[sorteddf['lat_bin'] == i]['lng'], q = 10, retbins = True)
    lng_bins.append(bins)
    lng_bins_edges.append(edges)
    
#insert and append edge values
for i in range(10):
    lng_bins_edges[i][0] = -180
    lng_bins_edges[i][10] = 180
    
#hard coding the latitude edges
#to a range format
lat_bins_edges = [(-90, -2.1667),
 (-2.1667, 15.4833),

 (31.4766, 34.4078),

#creating box objects for each 1% of

map_boxes = []
for i in range(10):
    for j in range(10):
        map_box = box(minx = lng_bins_edges[i][j] , miny = lat_bins_edges[i][0], maxx = lng_bins_edges[i][j+1], maxy = lat_bins_edges[i][1])
        map_boxes.append(map_box)
